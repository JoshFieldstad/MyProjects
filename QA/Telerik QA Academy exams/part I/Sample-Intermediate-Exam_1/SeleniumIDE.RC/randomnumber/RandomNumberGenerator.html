RandomNumberGenerator
open http://www.random.org/integers/
selectWindow null
waitForText css=h2 Random Integer Generator
verifyText css=h2 Random Integer Generator
type name=num 1
type name=max 2
clickAndWait //input[@value='Get Numbers']
selectWindow null
storeText css=pre.data fiRand
echo ${fiRand}
selectWindow null
refresh
selectWindow null
waitForPageToLoad 10000
storeText css=pre.data newRand
echo ${newRand}
selectWindow null
storeEval javascript{storedVars['fiRand']===storedVars['newRand']} c
echo ${c}
while storedVars['fiRand']==storedVars['newRand']
selectWindow null
refresh
waitForPageToLoad 10000
storeText css=pre.data newRand
echo ${newRand}
endWhile
